{ \
if (__builtin_constant_p(level) && (level) != 1) \
__asm__ volatile("cpuid\n\t" \
: "=a"(a), \
"=b"(b), \
"=c"(c), \
"=d"(d) \
: "0"(level)); \
else \
__asm__ volatile("cpuid\n\t" \
: "=a"(a), \

} while (0)
