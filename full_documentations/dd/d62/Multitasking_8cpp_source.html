263  int Priority, enum Architecture Architecture, enum Platform Platform)
264  {
265  EnterCriticalSection;
266  if (Parent == nullptr || Parent->Checksum != Checksum::PROCESS_CHECKSUM)
267  {
268  err("Null parent process not allowed.");
269  LeaveCriticalSection;
270  return nullptr;
271  }
272  TCB *thread = new TCB;
273  thread->Checksum = Checksum::THREAD_CHECKSUM;
274 
275  thread->ID = this->NextTID++;
276  thread->Status = STATUS::Ready;
277  thread->Security.Token = CreateToken();
278  thread->Parent = Parent;
279  memcpy(thread->Name, Parent->Name, sizeof(Parent->Name));
280  memset(&thread->Registers, 0, sizeof(TrapFrame));
281 
282  schedbg("Parent elevation is %d", Parent->Elevation);
283 
284  switch (Parent->Elevation)
285  {
286  case ELEVATION::System:
287  err("Elevation not supported. Using kernel elevation.");
288  [[fallthrough]];
289  case ELEVATION::Idle:
290  case ELEVATION::Kernel:
291  TrustToken(thread->Security.Token, false, thread->ID, TokenTrustLevel::TrustedByKernel);
292  thread->gs = (uint64_t)thread;
293  thread->fs = rdmsr(MSR_FS_BASE);
294  thread->Registers.cs = GDT_KERNEL_CODE;
295  thread->Registers.ss = GDT_KERNEL_DATA;
296  thread->Registers.rflags.always_one = 1;
297  thread->Registers.rflags.IF = 1;
298  thread->Registers.rflags.ID = 1;
299  thread->Stack = KernelStackAllocator->AllocateStack();
300  thread->Registers.STACK = (uint64_t)thread->Stack;
301  POKE(uint64_t, thread->Registers.rsp) = (uint64_t)ThreadDoExit;
302  break;
303  case ELEVATION::User:
304  TrustToken(thread->Security.Token, false, thread->ID, TokenTrustLevel::Untrusted);
305  thread->gs = 0;
306  thread->fs = rdmsr(MSR_FS_BASE);
307  thread->Registers.cs = GDT_USER_CODE;
308  thread->Registers.ss = GDT_USER_DATA;
309  thread->Registers.rflags.always_one = 1;
310  thread->Registers.rflags.IF = 1;
311  thread->Registers.rflags.ID = 1;
312  thread->Stack = KernelStackAllocator->AllocateStack(true);
313  thread->Registers.STACK = (uint64_t)thread->Stack;
314  /*
315  We can't add ThreadDoExit at the end of the stack because it will
316  trigger a page fault or invalid general protection fault exception (because
317  it will execute kernel code ofc).
318  We need to leave the libc's crt to make a syscall when the thread is exited.
319  */
320  break;
321  default:
322  err("Unknown elevation.");
323  delete thread;
324  LeaveCriticalSection;
325  return nullptr;
326  }
327 
328  thread->Registers.FUNCTION = (InstructionPointer + Parent->Offset);
329  thread->Registers.ARG0 = Arg0; // args0
330  thread->Registers.ARG1 = Arg1; // args1
331  thread->argc = Arg0;
332  thread->argv = (char **)Arg1;
333 
334  SetInfo(&thread->Info);
335  thread->Info.Architecture = Architecture;
336  thread->Info.Platform = Platform;
337  thread->Info.Priority = Priority;
338  Parent->Threads.push_back(thread);
339  trace("New thread %d created (%s).", thread->ID, thread->Name);
340  LeaveCriticalSection;
341  return thread;
342  }

608  break;
609  tmpidx++;
610  goto RetryAnotherThread;
611  }
612 
613  schedbg("%s(%d) and next thread is %s(%d)", CurrentCPU->CurrentProcess->Threads[i]->Name, CurrentCPU->CurrentProcess->Threads[i]->ID, thread->Name, thread->ID);
614 
615  // Check if the thread is ready to be executed.
616  if (thread->Status != STATUS::Ready)
617  {
618  schedbg("Thread %d is not ready", thread->ID);
619  goto RetryAnotherThread;
620  }
621 
622  // Everything is fine, we can set the new thread as the current one.
623  CurrentCPU->CurrentThread = thread;
624  schedbg("[thd 0 -> end] Scheduling thread %d parent of %s->%d Procs %d", thread->ID, thread->Parent->Name, CurrentCPU->CurrentProcess->Threads.size(), mt->ListProcess.size());
625  // Yay! We found a new thread to execute.
626  goto Success;
627  }
628  }
629 
630  // If the last process didn't find a thread to execute, we search for a new process.
631  for (uint64_t i = 0; i < mt->ListProcess.size(); i++)
632  {
633  // Loop until we find the current process from the process list.
634  if (mt->ListProcess[i] == CurrentCPU->CurrentProcess)
635  {
636  // Check if the next process is valid. If not, we search until we find.
637  uint64_t tmpidx = i;
638  RetryAnotherProcess:
639  PCB *pcb = mt->ListProcess[tmpidx + 1];
640  if (InvalidPCB(pcb))
641  {
642  if (tmpidx > mt->ListProcess.size())
643  break;
644  tmpidx++;
645  goto RetryAnotherProcess;
646  }
647 
648  if (pcb->Status != STATUS::Ready)
649  goto RetryAnotherProcess;
650 
651  // Everything good, now search for a thread.
652  for (uint64_t j = 0; j < pcb->Threads.size(); j++)
653  {
654  TCB *tcb = pcb->Threads[j];
655  if (InvalidTCB(tcb))
656  continue;
657  if (tcb->Status != STATUS::Ready)
658  continue;
659  // Success! We set as the current one and restore the stuff.
660  CurrentCPU->CurrentProcess = pcb;
661  CurrentCPU->CurrentThread = tcb;
662  schedbg("[cur proc+1 -> first thd] Scheduling thread %d %s->%d (Total Procs %d)", tcb->ID, tcb->Name, pcb->Threads.size(), mt->ListProcess.size());
663  goto Success;
664  }
665  }
666  }
667 
668  // Before checking from the beginning, we remove everything that is terminated.
669  foreach (PCB *pcb in mt->ListProcess)
670  {
671  if (InvalidPCB(pcb))
672  continue;
673  // RemoveProcess(pcb); // comment this until i will find a way to handle properly vectors, the memory need to be 0ed after removing.
674  }
675 
676  // If we didn't find anything, we check from the start of the list. This is the last chance to find something or we go to idle.
677  foreach (PCB *pcb in mt->ListProcess)
678  {
679  if (InvalidPCB(pcb))
680  continue;
681  if (pcb->Status != STATUS::Ready)
682  continue;
683 
684  // Now do the thread search!
685  foreach (TCB *tcb in pcb->Threads)
686  {
687  if (InvalidTCB(tcb))
688  continue;
689  if (tcb->Status != STATUS::Ready)
690  continue;
691  // \o/ We found a new thread to execute.
692  CurrentCPU->CurrentProcess = pcb;
693  CurrentCPU->CurrentThread = tcb;
694  schedbg("[proc 0 -> end -> first thd] Scheduling thread %d parent of %s->%d (Procs %d)", tcb->ID, tcb->Parent->Name, pcb->Threads.size(), mt->ListProcess.size());
695  goto Success;
696  }
697  }
698  }
699 
700  Idle:
701  {
702  // I should remove processes that are no longer having any threads? remove only from userspace?
703  if (mt->IdleProcess == nullptr)
704  {
705  schedbg("Idle process created");
706  mt->IdleProcess = mt->CreateProcess(nullptr, (char *)"idle", ELEVATION::Idle);
707  mt->IdleThread = mt->CreateThread(mt->IdleProcess, reinterpret_cast<uint64_t>(IdleProcessLoop), 0, 0);
708  }
709  CurrentCPU->CurrentProcess = mt->IdleProcess;
710  CurrentCPU->CurrentThread = mt->IdleThread;
711 
712  *regs = CurrentCPU->CurrentThread->Registers;
713  CR3 cr3;
714  cr3.raw = (uint64_t)KernelPageTableManager.PML4;
715  UpdatePageTable(cr3);
716 
717  wrmsr(MSR_FS_BASE, CurrentCPU->CurrentThread->fs);
718  wrmsr(MSR_GS_BASE, (uint64_t)CurrentCPU->CurrentThread);
719  wrmsr(MSR_SHADOW_GS_BASE, (uint64_t)CurrentCPU->CurrentThread);
720  _fxrstor(CurrentCPU->CurrentThread->FXRegion);
721  goto End;
722  }
723 
724  Success:
725  {
726  schedbg("Success Prc:%s(%d) Thd:%s(%d)->RIP:%#lx-RSP:%#lx(STACK: %#lx)",
727  CurrentCPU->CurrentProcess->Name, CurrentCPU->CurrentProcess->ID,
728  CurrentCPU->CurrentThread->Name, CurrentCPU->CurrentThread->ID,
729  CurrentCPU->CurrentThread->Registers.rip, CurrentCPU->CurrentThread->Registers.rsp, CurrentCPU->CurrentThread->Stack);
730

732  CurrentCPU->CurrentThread->Status = STATUS::Running;
733 
734  *regs = CurrentCPU->CurrentThread->Registers;
735  UpdatePageTable(CurrentCPU->CurrentProcess->PageTable);
736 
737  wrmsr(MSR_FS_BASE, CurrentCPU->CurrentThread->fs);
738  wrmsr(MSR_GS_BASE, (uint64_t)CurrentCPU->CurrentThread);
739  switch (CurrentCPU->CurrentProcess->Elevation)

741  case ELEVATION::System:
742  case ELEVATION::Idle:
743  case ELEVATION::Kernel:

751  break;
752  }
753  _fxrstor(CurrentCPU->CurrentThread->FXRegion);
754  }
755  End:

767  }
