287 int npf_parse_format_spec(char const *format, npf_format_spec_t *out_spec) {
288  char const *cur = format;
289 
290 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
291  out_spec->left_justified = 0;
292  out_spec->leading_zero_pad = 0;
293 #endif
294  out_spec->case_adjust = 'a'-'A'; // lowercase
295  out_spec->prepend = 0;
296  out_spec->alt_form = 0;
297 
298  while (*++cur) { // cur points at the leading '%' character
299  switch (*cur) { // Optional flags
300 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
301  case '-':
302  out_spec->left_justified = '-';
303  out_spec->leading_zero_pad = 0;
304  continue;
305  case '0':
306  out_spec->leading_zero_pad = !out_spec->left_justified;
307  continue;
308 #endif
309  case '+':
310  out_spec->prepend = '+';
311  continue;
312  case ' ':
313  if (out_spec->prepend == 0) { out_spec->prepend = ' '; }
314  continue;
315  case '#':
316  out_spec->alt_form = '#';
317  continue;
318  default: break;
319  }
320  break;
321  }
322 
323 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
324  out_spec->field_width_opt = NPF_FMT_SPEC_OPT_NONE;
325  if (*cur == '*') {
326  out_spec->field_width_opt = NPF_FMT_SPEC_OPT_STAR;
327  ++cur;
328  } else {
329  out_spec->field_width = 0;
330  while ((*cur >= '0') && (*cur <= '9')) {
331  out_spec->field_width_opt = NPF_FMT_SPEC_OPT_LITERAL;
332  out_spec->field_width = (out_spec->field_width * 10) + (*cur++ - '0');
333  }
334  }
335 #endif
336 
337 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
338  out_spec->prec = 0;
339  out_spec->prec_opt = NPF_FMT_SPEC_OPT_NONE;
340  if (*cur == '.') {
341  ++cur;
342  if (*cur == '*') {
343  out_spec->prec_opt = NPF_FMT_SPEC_OPT_STAR;
344  ++cur;
345  } else {
346  if (*cur == '-') {
347  ++cur;
348  out_spec->prec_opt = NPF_FMT_SPEC_OPT_NONE;
349  } else {
350  out_spec->prec_opt = NPF_FMT_SPEC_OPT_LITERAL;
351  }
352  while ((*cur >= '0') && (*cur <= '9')) {
353  out_spec->prec = (out_spec->prec * 10) + (*cur++ - '0');
354  }
355  }
356  }
357 #endif
358 
359  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_NONE;
360  switch (*cur++) { // Length modifier
361  case 'h':
362  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_SHORT;
363  if (*cur == 'h') {
364  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_CHAR;
365  ++cur;
366  }
367  break;
368  case 'l':
369  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_LONG;
370 #if NANOPRINTF_USE_LARGE_FORMAT_SPECIFIERS == 1
371  if (*cur == 'l') {
372  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_LARGE_LONG_LONG;
373  ++cur;
374  }
375 #endif
376  break;
377 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
378  case 'L':
379  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_LONG_DOUBLE;
380  break;
381 #endif
382 #if NANOPRINTF_USE_LARGE_FORMAT_SPECIFIERS == 1
383  case 'j':
384  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_LARGE_INTMAX;
385  break;
386  case 'z':
387  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_LARGE_SIZET;
388  break;
389  case 't':
390  out_spec->length_modifier = NPF_FMT_SPEC_LEN_MOD_LARGE_PTRDIFFT;
391  break;
392 #endif
393  default: --cur; break;
394  }
395 
396  switch (*cur++) { // Conversion specifier
397  case '%':
398  out_spec->conv_spec = NPF_FMT_SPEC_CONV_PERCENT;
399 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
400  out_spec->prec_opt = NPF_FMT_SPEC_OPT_NONE;
401 #endif
402  break;
403  case 'c':
404  out_spec->conv_spec = NPF_FMT_SPEC_CONV_CHAR;
405 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
406  out_spec->prec_opt = NPF_FMT_SPEC_OPT_NONE;
407 #endif
408  break;
409  case 's':
410  out_spec->conv_spec = NPF_FMT_SPEC_CONV_STRING;
411 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
412  out_spec->leading_zero_pad = 0;
413 #endif
414  break;
415 
416  case 'i':
417  case 'd':
418  out_spec->conv_spec = NPF_FMT_SPEC_CONV_SIGNED_INT;
419  break;
420 
421  case 'o':
422  out_spec->conv_spec = NPF_FMT_SPEC_CONV_OCTAL;
423  break;
424  case 'u':
425  out_spec->conv_spec = NPF_FMT_SPEC_CONV_UNSIGNED_INT;
426  break;
427 
428  case 'X':
429  out_spec->case_adjust = 0;
430  case 'x':
431  out_spec->conv_spec = NPF_FMT_SPEC_CONV_HEX_INT;
432  break;
433 
434 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
435  case 'F':
436  out_spec->case_adjust = 0;
437  case 'f':
438  out_spec->conv_spec = NPF_FMT_SPEC_CONV_FLOAT_DECIMAL;
439  if (out_spec->prec_opt == NPF_FMT_SPEC_OPT_NONE) { out_spec->prec = 6; }
440  break;
441 #endif
442 
443 #if NANOPRINTF_USE_WRITEBACK_FORMAT_SPECIFIERS == 1
444  case 'n':
445  // todo: reject string if flags or width or precision exist
446  out_spec->conv_spec = NPF_FMT_SPEC_CONV_WRITEBACK;
447 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
448  out_spec->prec_opt = NPF_FMT_SPEC_OPT_NONE;
449 #endif
450  break;
451 #endif
452 
453  case 'p':
454  out_spec->conv_spec = NPF_FMT_SPEC_CONV_POINTER;
455 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
456  out_spec->prec_opt = NPF_FMT_SPEC_OPT_NONE;
457 #endif
458  break;
459 
460 #if NANOPRINTF_USE_BINARY_FORMAT_SPECIFIERS == 1
461  case 'B':
462  out_spec->case_adjust = 0;
463  case 'b':
464  out_spec->conv_spec = NPF_FMT_SPEC_CONV_BINARY;
465  break;
466 #endif
467 
468  default: return 0;
469  }
470 
471  return (int)(cur - format);
472 }

501 int npf_fsplit_abs(float f, uint64_t *out_int_part, uint64_t *out_frac_part,
502  int *out_frac_base10_neg_exp) {
503  /* conversion algorithm by Wojciech Muła (zdjęcia@garnek.pl)
504  http://0x80.pl/notesen/2015-12-29-float-to-string.html
505  grisu2 (https://bit.ly/2JgMggX) and ryu (https://bit.ly/2RLXSg0)
506  are fast + precise + round, but require large lookup tables. */
507 
508  uint32_t f_bits; { // union-cast is UB, let compiler optimize byte-copy loop.
509  char const *src = (char const *)&f;
510  char *dst = (char *)&f_bits;
511  for (unsigned i = 0; i < sizeof(f_bits); ++i) { dst[i] = src[i]; }
512  }
513 
514  int const exponent =
515  ((int)((f_bits >> NPF_MANTISSA_BITS) & ((1u << NPF_EXPONENT_BITS) - 1u)) -
516  NPF_EXPONENT_BIAS) - NPF_MANTISSA_BITS;
517 
518  if (exponent >= (64 - NPF_MANTISSA_BITS)) { return 0; } // value is out of range
519 
520  uint32_t const implicit_one = 1u << NPF_MANTISSA_BITS;
521  uint32_t const mantissa = f_bits & (implicit_one - 1);
522  uint32_t const mantissa_norm = mantissa | implicit_one;
523 
524  if (exponent > 0) {
525  *out_int_part = (uint64_t)mantissa_norm << exponent;
526  } else if (exponent < 0) {
527  if (-exponent > NPF_MANTISSA_BITS) {
528  *out_int_part = 0;
529  } else {
530  *out_int_part = mantissa_norm >> -exponent;
531  }
532  } else {
533  *out_int_part = mantissa_norm;
534  }
535 
536  uint64_t frac; {
537  int const shift = NPF_FRACTION_BIN_DIGITS + exponent - 4;
538  if ((shift >= (NPF_FRACTION_BIN_DIGITS - 4)) || (shift < 0)) {
539  frac = 0;
540  } else {
541  frac = ((uint64_t)mantissa_norm) << shift;
542  }
543  // multiply off the leading one's digit
544  frac &= 0x0fffffffffffffffllu;
545  frac *= 10;
546  }
547 
548  { // Count the number of 0s at the beginning of the fractional part.
549  int frac_base10_neg_exp = 0;
550  while (frac && ((frac >> (NPF_FRACTION_BIN_DIGITS - 4))) == 0) {
551  ++frac_base10_neg_exp;
552  frac &= 0x0fffffffffffffffllu;
553  frac *= 10;
554  }
555  *out_frac_base10_neg_exp = frac_base10_neg_exp;
556  }
557 
558  { // Convert the fractional part to base 10.
559  uint64_t frac_part = 0;
560  for (int i = 0; frac && (i < NPF_MAX_FRACTION_DEC_DIGITS); ++i) {
561  frac_part *= 10;
562  frac_part += (uint64_t)(frac >> (NPF_FRACTION_BIN_DIGITS - 4));
563  frac &= 0x0fffffffffffffffllu;
564  frac *= 10;
565  }
566  *out_frac_part = frac_part;
567  }
568  return 1;
569 }
570 
571 int npf_ftoa_rev(char *buf, float f, char case_adj, int *out_frac_chars) {
572  uint32_t f_bits; { // union-cast is UB, let compiler optimize byte-copy loop.
573  char const *src = (char const *)&f;
574  char *dst = (char *)&f_bits;
575  for (unsigned i = 0; i < sizeof(f_bits); ++i) { dst[i] = src[i]; }
576  }
577 
578  if ((uint8_t)(f_bits >> 23) == 0xFF) {
579  if (f_bits & 0x7fffff) {
580  for (int i = 0; i < 3; ++i) { *buf++ = (char)("NAN"[i] + case_adj); }
581  } else {
582  for (int i = 0; i < 3; ++i) { *buf++ = (char)("FNI"[i] + case_adj); }
583  }
584  return -3;
585  }
586 
587  uint64_t int_part, frac_part;
588  int frac_base10_neg_exp;
589  if (npf_fsplit_abs(f, &int_part, &frac_part, &frac_base10_neg_exp) == 0) {
590  for (int i = 0; i < 3; ++i) { *buf++ = (char)("ROO"[i] + case_adj); }
591  return -3;
592  }
593 
594  char *dst = buf;
595 
596  while (frac_part) { // write the fractional digits
597  *dst++ = (char)('0' + (frac_part % 10));
598  frac_part /= 10;
599  }
600 
601  // write the 0 digits between the . and the first fractional digit
602  while (frac_base10_neg_exp-- > 0) { *dst++ = '0'; }
603  *out_frac_chars = (int)(dst - buf);
604  *dst++ = '.';
605 
606  // write the integer digits
607  do { *dst++ = (char)('0' + (int_part % 10)); int_part /= 10; } while (int_part);
608  return (int)(dst - buf);
609 }
610 
611 #endif // NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS

616  if (!u) { return 1; }

667  ++pc_cnt->n;

671 #define NPF_PUTC(VAL) do { npf_putc_cnt((int)(VAL), &pc_cnt); } while (0)
672 
673 #define NPF_EXTRACT(MOD, CAST_TO, EXTRACT_AS) \

679 int npf_vpprintf(npf_putc pc, void *pc_ctx, char const *format, va_list args) {
680  npf_format_spec_t fs;
681  char const *cur = format;
682  npf_cnt_putc_ctx_t pc_cnt;
683  pc_cnt.pc = pc;
684  pc_cnt.ctx = pc_ctx;
685  pc_cnt.n = 0;
686 
687  while (*cur) {
688  int const fs_len = (*cur != '%') ? 0 : npf_parse_format_spec(cur, &fs);
689  if (!fs_len) { NPF_PUTC(*cur++); continue; }
690  cur += fs_len;
691 
692  // Extract star-args immediately
693 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
694  if (fs.field_width_opt == NPF_FMT_SPEC_OPT_STAR) {
695  fs.field_width_opt = NPF_FMT_SPEC_OPT_LITERAL;
696  fs.field_width = va_arg(args, int);
697  if (fs.field_width < 0) {
698  fs.field_width = -fs.field_width;
699  fs.left_justified = 1;
700  }
701  }
702 #endif
703 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
704  if (fs.prec_opt == NPF_FMT_SPEC_OPT_STAR) {
705  fs.prec_opt = NPF_FMT_SPEC_OPT_NONE;
706  fs.prec = va_arg(args, int);
707  if (fs.prec >= 0) { fs.prec_opt = NPF_FMT_SPEC_OPT_LITERAL; }
708  }
709 #endif
710 
711  union { char cbuf_mem[32]; npf_uint_t binval; } u;
712  char *cbuf = u.cbuf_mem, sign_c = 0;
713  int cbuf_len = 0, need_0x = 0;
714 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
715  int field_pad = 0;
716  char pad_c = 0;
717 #endif
718 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
719  int prec_pad = 0;
720 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
721  int zero = 0;
722 #endif
723 #endif
724 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
725  int frac_chars = 0, inf_or_nan = 0;
726 #endif
727 
728  // Extract and convert the argument to string, point cbuf at the text.
729  switch (fs.conv_spec) {
730  case NPF_FMT_SPEC_CONV_PERCENT:
731  *cbuf = '%';
732  ++cbuf_len;
733  break;
734 
735  case NPF_FMT_SPEC_CONV_CHAR:
736  *cbuf = (char)va_arg(args, int);
737  ++cbuf_len;
738  break;
739 
740  case NPF_FMT_SPEC_CONV_STRING: {
741  cbuf = va_arg(args, char *);
742  for (char const *s = cbuf; *s; ++s, ++cbuf_len); // strlen
743 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
744  if (fs.prec_opt == NPF_FMT_SPEC_OPT_LITERAL) {
745  cbuf_len = npf_min(fs.prec, cbuf_len); // prec truncates strings
746  }
747 #endif
748  } break;
749 
750  case NPF_FMT_SPEC_CONV_SIGNED_INT: {
751  npf_int_t val = 0;
752  switch (fs.length_modifier) {
753  NPF_EXTRACT(NONE, int, int);
754  NPF_EXTRACT(SHORT, short, int);
755  NPF_EXTRACT(LONG_DOUBLE, int, int);
756  NPF_EXTRACT(CHAR, char, int);
757  NPF_EXTRACT(LONG, long, long);
758 #if NANOPRINTF_USE_LARGE_FORMAT_SPECIFIERS == 1
759  NPF_EXTRACT(LARGE_LONG_LONG, long long, long long);
760  NPF_EXTRACT(LARGE_INTMAX, intmax_t, intmax_t);
761  NPF_EXTRACT(LARGE_SIZET, ssize_t, ssize_t);
762  NPF_EXTRACT(LARGE_PTRDIFFT, ptrdiff_t, ptrdiff_t);
763 #endif
764  default: break;
765  }
766 
767  sign_c = (val < 0) ? '-' : fs.prepend;
768 
769 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
770 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
771  zero = !val;
772 #endif
773  // special case, if prec and value are 0, skip
774  if (!val && (fs.prec_opt == NPF_FMT_SPEC_OPT_LITERAL) && !fs.prec) {
775  cbuf_len = 0;
776  } else
777 #endif
778  { cbuf_len = npf_itoa_rev(cbuf, val); }
779  } break;
780 
781 #if NANOPRINTF_USE_BINARY_FORMAT_SPECIFIERS == 1
782  case NPF_FMT_SPEC_CONV_BINARY:
783 #endif
784  case NPF_FMT_SPEC_CONV_OCTAL:
785  case NPF_FMT_SPEC_CONV_HEX_INT:
786  case NPF_FMT_SPEC_CONV_UNSIGNED_INT: {
787  npf_uint_t val = 0;
788 
789  switch (fs.length_modifier) {
790  NPF_EXTRACT(NONE, unsigned, unsigned);
791  NPF_EXTRACT(SHORT, unsigned short, unsigned);
792  NPF_EXTRACT(LONG_DOUBLE, unsigned, unsigned);
793  NPF_EXTRACT(CHAR, unsigned char, unsigned);
794  NPF_EXTRACT(LONG, unsigned long, unsigned long);
795 #if NANOPRINTF_USE_LARGE_FORMAT_SPECIFIERS == 1
796  NPF_EXTRACT(LARGE_LONG_LONG, unsigned long long, unsigned long long);
797  NPF_EXTRACT(LARGE_INTMAX, uintmax_t, uintmax_t);
798  NPF_EXTRACT(LARGE_SIZET, size_t, size_t);
799  NPF_EXTRACT(LARGE_PTRDIFFT, size_t, size_t);
800 #endif
801  default: break;
802  }
803 
804 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
805 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
806  zero = !val;
807 #endif
808  if (!val && (fs.prec_opt == NPF_FMT_SPEC_OPT_LITERAL) && !fs.prec) {
809  // Zero value and explicitly-requested zero precision means "print nothing".
810  if ((fs.conv_spec == NPF_FMT_SPEC_CONV_OCTAL) && fs.alt_form) {
811  fs.prec = 1; // octal special case, print a single '0'
812  }
813  } else
814 #endif
815 #if NANOPRINTF_USE_BINARY_FORMAT_SPECIFIERS == 1
816  if (fs.conv_spec == NPF_FMT_SPEC_CONV_BINARY) {
817  cbuf_len = npf_bin_len(val); u.binval = val;
818  } else
819 #endif
820  {
821  unsigned const base = (fs.conv_spec == NPF_FMT_SPEC_CONV_OCTAL) ?
822  8u : ((fs.conv_spec == NPF_FMT_SPEC_CONV_HEX_INT) ? 16u : 10u);
823  cbuf_len = npf_utoa_rev(cbuf, val, base, (unsigned)fs.case_adjust);
824  }
825 
826  if (val && fs.alt_form && (fs.conv_spec == NPF_FMT_SPEC_CONV_OCTAL)) {
827  cbuf[cbuf_len++] = '0'; // OK to add leading octal '0' immediately.
828  }
829 
830  if (val && fs.alt_form) { // 0x or 0b but can't write it yet.
831  if (fs.conv_spec == NPF_FMT_SPEC_CONV_HEX_INT) { need_0x = 'X'; }
832 #if NANOPRINTF_USE_BINARY_FORMAT_SPECIFIERS == 1
833  else if (fs.conv_spec == NPF_FMT_SPEC_CONV_BINARY) { need_0x = 'B'; }
834 #endif
835  if (need_0x) { need_0x += fs.case_adjust; }
836  }
837  } break;
838 
839  case NPF_FMT_SPEC_CONV_POINTER: {
840  cbuf_len =
841  npf_utoa_rev(cbuf, (npf_uint_t)(uintptr_t)va_arg(args, void *), 16, 'a'-'A');
842  need_0x = 'x';
843  } break;
844 
845 #if NANOPRINTF_USE_WRITEBACK_FORMAT_SPECIFIERS == 1
846  case NPF_FMT_SPEC_CONV_WRITEBACK:
847  switch (fs.length_modifier) {
848  NPF_WRITEBACK(NONE, int);
849  NPF_WRITEBACK(SHORT, short);
850  NPF_WRITEBACK(LONG, long);
851  NPF_WRITEBACK(LONG_DOUBLE, double);
852  NPF_WRITEBACK(CHAR, signed char);
853 #if NANOPRINTF_USE_LARGE_FORMAT_SPECIFIERS == 1
854  NPF_WRITEBACK(LARGE_LONG_LONG, long long);
855  NPF_WRITEBACK(LARGE_INTMAX, intmax_t);
856  NPF_WRITEBACK(LARGE_SIZET, size_t);
857  NPF_WRITEBACK(LARGE_PTRDIFFT, ptrdiff_t);
858 #endif
859  default: break;
860  } break;
861 #endif
862 
863 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
864  case NPF_FMT_SPEC_CONV_FLOAT_DECIMAL: {
865  float val;
866  if (fs.length_modifier == NPF_FMT_SPEC_LEN_MOD_LONG_DOUBLE) {
867  val = (float)va_arg(args, long double);
868  } else {
869  val = (float)va_arg(args, double);
870  }
871 
872  sign_c = (val < 0.f) ? '-' : fs.prepend;
873 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
874  zero = (val == 0.f);
875 #endif
876  cbuf_len = npf_ftoa_rev(cbuf, val, fs.case_adjust, &frac_chars);
877 
878  if (cbuf_len < 0) {
879  cbuf_len = -cbuf_len;
880  inf_or_nan = 1;
881  } else {
882  int const prec_adj = npf_max(0, frac_chars - fs.prec);
883  cbuf += prec_adj;
884  cbuf_len -= prec_adj;
885  }
886  } break;
887 #endif
888  default: break;
889  }
890 
891 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
892  // Compute the field width pad character
893  if (fs.field_width_opt == NPF_FMT_SPEC_OPT_LITERAL) {
894  if (fs.leading_zero_pad) { // '0' flag is only legal with numeric types
895  if ((fs.conv_spec != NPF_FMT_SPEC_CONV_STRING) &&
896  (fs.conv_spec != NPF_FMT_SPEC_CONV_CHAR) &&
897  (fs.conv_spec != NPF_FMT_SPEC_CONV_PERCENT)) {
898 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
899  if ((fs.prec_opt == NPF_FMT_SPEC_OPT_LITERAL) && !fs.prec && zero) {
900  pad_c = ' ';
901  } else
902 #endif
903  { pad_c = '0'; }
904  }
905  } else { pad_c = ' '; }
906  }
907 #endif
908 
909  // Compute the number of bytes to truncate or '0'-pad.
910  if (fs.conv_spec != NPF_FMT_SPEC_CONV_STRING) {
911 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
912  if (!inf_or_nan) { // float precision is after the decimal point
913  int const prec_start =
914  (fs.conv_spec == NPF_FMT_SPEC_CONV_FLOAT_DECIMAL) ? frac_chars : cbuf_len;
915  prec_pad = npf_max(0, fs.prec - prec_start);
916  }
917 #elif NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
918  prec_pad = npf_max(0, fs.prec - cbuf_len);
919 #endif
920  }
921 
922 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
923  // Given the full converted length, how many pad bytes?
924  field_pad = fs.field_width - cbuf_len - !!sign_c;
925  if (need_0x) { field_pad -= 2; }
926 
927 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
928  if ((fs.conv_spec == NPF_FMT_SPEC_CONV_FLOAT_DECIMAL) && !fs.prec && !fs.alt_form) {
929  ++field_pad; // 0-pad, no decimal point.
930  }
931 #endif
932 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
933  field_pad -= prec_pad;
934 #endif
935  field_pad = npf_max(0, field_pad);
936 #endif // NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS
937 
938 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
939  // Apply right-justified field width if requested
940  if (!fs.left_justified && pad_c) { // If leading zeros pad, sign goes first.
941  if (pad_c == '0') {
942  if (sign_c) { NPF_PUTC(sign_c); sign_c = 0; }
943  // Pad byte is '0', write '0x' before '0' pad chars.
944  if (need_0x) { NPF_PUTC('0'); NPF_PUTC(need_0x); }
945  }
946  while (field_pad-- > 0) { NPF_PUTC(pad_c); }
947  // Pad byte is ' ', write '0x' after ' ' pad chars but before number.
948  if ((pad_c != '0') && need_0x) { NPF_PUTC('0'); NPF_PUTC(need_0x); }
949  } else
950 #endif
951  { if (need_0x) { NPF_PUTC('0'); NPF_PUTC(need_0x); } } // no pad, '0x' requested.
952 
953  // Write the converted payload
954  if (fs.conv_spec == NPF_FMT_SPEC_CONV_STRING) {
955  for (int i = 0; i < cbuf_len; ++i) { NPF_PUTC(cbuf[i]); }
956  } else {
957  if (sign_c) { NPF_PUTC(sign_c); }
958 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
959  if (fs.conv_spec != NPF_FMT_SPEC_CONV_FLOAT_DECIMAL) {
960 #endif
961 
962 #if NANOPRINTF_USE_PRECISION_FORMAT_SPECIFIERS == 1
963  while (prec_pad-- > 0) { NPF_PUTC('0'); } // int precision leads.
964 #endif
965 
966 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
967  } else {
968  // if 0 precision, skip the fractional part and '.'
969  // if 0 prec + alternative form, keep the '.'
970  if (!fs.prec && !fs.alt_form) { ++cbuf; --cbuf_len; }
971  }
972 #endif
973 
974 #if NANOPRINTF_USE_BINARY_FORMAT_SPECIFIERS == 1
975  if (fs.conv_spec == NPF_FMT_SPEC_CONV_BINARY) {
976  while (cbuf_len) { NPF_PUTC('0' + ((u.binval >> --cbuf_len) & 1)); }
977  } else
978 #endif
979  { while (cbuf_len-- > 0) { NPF_PUTC(cbuf[cbuf_len]); } } // payload is reversed
980 
981 #if NANOPRINTF_USE_FLOAT_FORMAT_SPECIFIERS == 1
982  // real precision comes after the number.
983  if ((fs.conv_spec == NPF_FMT_SPEC_CONV_FLOAT_DECIMAL) && !inf_or_nan) {
984  while (prec_pad-- > 0) { NPF_PUTC('0'); }
985  }
986 #endif
987  }
988 
989 #if NANOPRINTF_USE_FIELD_WIDTH_FORMAT_SPECIFIERS == 1
990  if (fs.left_justified && pad_c) { // Apply left-justified field width
991  while (field_pad-- > 0) { NPF_PUTC(pad_c); }
992  }
993 #endif
994  }
995 
996  return pc_cnt.n;
997 }
