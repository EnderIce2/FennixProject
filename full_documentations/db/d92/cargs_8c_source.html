180 static void cag_option_parse_value(cag_option_context *context,
181  const cag_option *option, char **c)
182 {
183  // And now let's check whether this option is supposed to have a value, which
184  // is the case if there is a value name set. The value can be either submitted
185  // with a '=' sign or a space, which means we would have to jump over to the
186  // next argv index. This is somewhat ugly, but we do it to behave the same as
187  // the other option parsers.
188  if (option->value_name != NULL) {
189  if (**c == '=') {
190  context->value = ++(*c);
191  } else {
192  // If the next index is larger or equal to the argument count, then the
193  // parameter for this option is missing. The user will know about this,
194  // since the value pointer of the context will be NULL because we don't
195  // set it here in that case.
196  if (context->argc > context->index + 1) {
197  // We consider this argv to be the value, no matter what the contents
198  // are.
199  ++context->index;
200  *c = context->argv[context->index];
201  context->value = *c;
202  }
203  }
204 
205  // Move c to the end of the value, to not confuse the caller about our
206  // position.
207  while (**c) {
208  ++(*c);
209  }
210  }

213 static void cag_option_parse_access_name(cag_option_context *context, char **c)
214 {
215  const cag_option *option;
216  char *n;
217 
218  // Now we need to extract the access name, which is any symbol up to a '=' or
219  // a '\0'.
220  n = *c;
221  while (**c && **c != '=') {
222  ++*c;
223  }
224 
225  // Now this will obviously always be true, but we are paranoid. Sometimes. It
226  // doesn't hurt to check.
227  assert(*c >= n);
228 
229  // Figure out which option this name belongs to. This might return NULL if the
230  // name is not registered, which means the user supplied an unknown option. In
231  // that case we return true to indicate that we finished with this option. We
232  // have to skip the value parsing since we don't know whether the user thinks
233  // this option has one or not. Since we don't set any identifier specifically,
234  // it will remain '?' within the context.
235  option = cag_option_find_by_name(context, n, (size_t)(*c - n));
236  if (option == NULL) {
237  // Since this option is invalid, we will move on to the next index. There is
238  // nothing we can do about this.
239  ++context->index;
240  return;
241  }
242 
243  // We found an option and now we can specify the identifier within the
244  // context.
245  context->identifier = option->identifier;
246 
247  // And now we try to parse the value. This function will also check whether
248  // this option is actually supposed to have a value.
249  cag_option_parse_value(context, option, c);
250 
251  // And finally we move on to the next index.
252  ++context->index;

291 static void cag_option_shift(cag_option_context *context, int start, int option,
292  int end)
293 {
294  char *tmp;
295  int a_index, shift_index, shift_count, left_index, right_index;
296 
297  shift_count = option - start;
298 
299  // There is no shift is required if the start and the option have the same
300  // index.
301  if (shift_count == 0) {
302  return;
303  }
304 
305  // Lets loop through the option strings first, which we will move towards the
306  // beginning.
307  for (a_index = option; a_index < end; ++a_index) {
308  // First remember the current option value, because we will have to save
309  // that later at the beginning.
310  tmp = context->argv[a_index];
311 
312  // Let's loop over all option values and shift them one towards the end.
313  // This will override the option value we just stored temporarily.
314  for (shift_index = 0; shift_index < shift_count; ++shift_index) {
315  left_index = a_index - shift_index;
316  right_index = a_index - shift_index - 1;
317  context->argv[left_index] = context->argv[right_index];
318  }
319 
320  // Now restore the saved option value at the beginning.
321  context->argv[a_index - shift_count] = tmp;
322  }
323 
324  // The new index will be before all non-option values, in such a way that they
325  // all will be moved again in the next fetch call.
326  context->index = end - shift_count;
327 }

335 {

338

342

367 bool cag_option_fetch(cag_option_context *context)
368 {
369  char *c;
370  int old_index, new_index;
371 
372  // Reset our identifier to a question mark, which indicates an "unknown"
373  // option. The value is set to NULL, to make sure we are not carrying the
374  // parameter from the previous option to this one.
375  context->identifier = '?';
376  context->value = NULL;
377 
378  // Check whether there are any options left to parse and remember the old
379  // index as well as the new index. In the end we will move the option junk to
380  // the beginning, so that non option arguments can be read.
381  old_index = context->index;
382  new_index = cag_option_find_next(context);
383  if (new_index >= 0) {
384  context->index = new_index;
385  } else {
386  return false;
387  }
388 
389  // Grab a pointer to the beginning of the option. At this point, the next
390  // character must be a '-', since if it was not the prepare function would
391  // have returned false. We will skip that symbol and proceed.
392  c = context->argv[context->index];
393  assert(*c == '-');
394  ++c;
395 
396  // Check whether this is a long option, starting with a double "--".
397  if (*c == '-') {
398  ++c;
399 
400  // This might be a double "--" which indicates the end of options. If this
401  // is the case, we will not move to the next index. That ensures that
402  // another call to the fetch function will not skip the "--".
403  if (*c == '\0') {
404  context->forced_end = true;
405  } else {
406  // We parse now the access name. All information about it will be written
407  // to the context.
408  cag_option_parse_access_name(context, &c);
409  }
410  } else {
411  // This is no long option, so we can just parse an access letter.
412  cag_option_parse_access_letter(context, &c);
413  }
414 
415  // Move the items so that the options come first followed by non-option
416  // arguments.
417  cag_option_shift(context, old_index, new_index, context->index);
418 
419  return context->forced_end == false;
420 }
