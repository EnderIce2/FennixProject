577 // into its base-10 parts: integral - before the decimal point, and fractional - after it.
578 // Taken the precision into account, but does not change it even internally.

580 {

583  double abs_number = (number_.is_negative) ? -number : number;
584  number_.integral = (int_fast64_t)abs_number;
585  double remainder = (abs_number - (double) number_.integral) * powers_of_10[precision];

594  number_.fractional = 0;
595  ++number_.integral;
596  }
597  }

601  }

632  struct scaling_factor result;
633  if (sf.multiply) {
634  result.multiply = true;
635  result.raw_factor = sf.raw_factor * extra_multiplicative_factor;
636  }
637  else {
638  int factor_exp2 = get_exp2(get_bit_access(sf.raw_factor));
639  int extra_factor_exp2 = get_exp2(get_bit_access(extra_multiplicative_factor));
640 
641  // Divide the larger-exponent raw raw_factor by the smaller
642  if (PRINTF_ABS(factor_exp2) > PRINTF_ABS(extra_factor_exp2)) {
643  result.multiply = false;
644  result.raw_factor = sf.raw_factor / extra_multiplicative_factor;
645  }
646  else {
647  result.multiply = true;
648  result.raw_factor = extra_multiplicative_factor / sf.raw_factor;
649  }
650  }
651  return result;
652 }

654 static struct double_components get_normalized_components(bool negative, printf_size_t precision, double non_normalized, struct scaling_factor normalization, int floored_exp10)
655 {
656  struct double_components components;
657  components.is_negative = negative;
658  double scaled = apply_scaling(non_normalized, normalization);
659 
660  bool close_to_representation_extremum = ( (-floored_exp10 + (int) precision) >= DBL_MAX_10_EXP - 1 );
661  if (close_to_representation_extremum) {
662  // We can't have a normalization factor which also accounts for the precision, i.e. moves
663  // some decimal digits into the mantissa, since it's unrepresentable, or nearly unrepresentable.
664  // So, we'll give up early on getting extra precision...
665  return get_components(negative ? -scaled : scaled, precision);
666  }
667  components.integral = (int_fast64_t) scaled;
668  double remainder = non_normalized - unapply_scaling((double) components.integral, normalization);
669  double prec_power_of_10 = powers_of_10[precision];
670  struct scaling_factor account_for_precision = update_normalization(normalization, prec_power_of_10);
671  double scaled_remainder = apply_scaling(remainder, account_for_precision);
672  double rounding_threshold = 0.5;
673 
674  components.fractional = (int_fast64_t) scaled_remainder; // when precision == 0, the assigned value should be 0
675  scaled_remainder -= (double) components.fractional; //when precision == 0, this will not change scaled_remainder
676 
677  components.fractional += (scaled_remainder >= rounding_threshold);
678  if (scaled_remainder == rounding_threshold) {
679  // banker's rounding: Round towards the even number (making the mean error 0)
680  components.fractional &= ~((int_fast64_t) 0x1);
681  }
682  // handle rollover, e.g. the case of 0.99 with precision 1 becoming (0,100),
683  // and must then be corrected into (1, 0).
684  // Note: for precision = 0, this will "translate" the rounding effect from
685  // the fractional part to the integral part where it should actually be
686  // felt (as prec_power_of_10 is 1)
687  if ((double) components.fractional >= prec_power_of_10) {
688  components.fractional = 0;
689  ++components.integral;
690  }
691  return components;
692 }

695 static void print_broken_up_decimal(
696  struct double_components number_, output_gadget_t* output, printf_size_t precision,
697  printf_size_t width, printf_flags_t flags, char *buf, printf_size_t len)
698 {
699  if (precision != 0U) {
700  // do fractional part, as an unsigned number
701 
702  printf_size_t count = precision;
703 
704  // %g/%G mandates we skip the trailing 0 digits...
705  if ((flags & FLAGS_ADAPT_EXP) && !(flags & FLAGS_HASH) && (number_.fractional > 0)) {
706  while(true) {
707  int_fast64_t digit = number_.fractional % 10U;
708  if (digit != 0) {
709  break;
710  }
711  --count;
712  number_.fractional /= 10U;
713 
714  }
715  // ... and even the decimal point if there are no
716  // non-zero fractional part digits (see below)
717  }
718 
719  if (number_.fractional > 0 || !(flags & FLAGS_ADAPT_EXP) || (flags & FLAGS_HASH) ) {
720  while (len < PRINTF_DECIMAL_BUFFER_SIZE) {
721  --count;
722  buf[len++] = (char)('0' + number_.fractional % 10U);
723  if (!(number_.fractional /= 10U)) {
724  break;
725  }
726  }
727  // add extra 0s
728  while ((len < PRINTF_DECIMAL_BUFFER_SIZE) && (count > 0U)) {
729  buf[len++] = '0';
730  --count;
731  }
732  if (len < PRINTF_DECIMAL_BUFFER_SIZE) {
733  buf[len++] = '.';
734  }
735  }
736  }
737  else {
738  if ((flags & FLAGS_HASH) && (len < PRINTF_DECIMAL_BUFFER_SIZE)) {
739  buf[len++] = '.';
740  }
741  }
742 
743  // Write the integer part of the number (it comes after the fractional
744  // since the character order is reversed)
745  while (len < PRINTF_DECIMAL_BUFFER_SIZE) {
746  buf[len++] = (char)('0' + (number_.integral % 10));
747  if (!(number_.integral /= 10)) {
748  break;
749  }
750  }
751 
752  // pad leading zeros
753  if (!(flags & FLAGS_LEFT) && (flags & FLAGS_ZEROPAD)) {
754  if (width && (number_.is_negative || (flags & (FLAGS_PLUS | FLAGS_SPACE)))) {
755  width--;
756  }
757  while ((len < width) && (len < PRINTF_DECIMAL_BUFFER_SIZE)) {
758  buf[len++] = '0';
759  }
760  }
761 
762  if (len < PRINTF_DECIMAL_BUFFER_SIZE) {
763  if (number_.is_negative) {
764  buf[len++] = '-';
765  }
766  else if (flags & FLAGS_PLUS) {
767  buf[len++] = '+'; // ignore the space if the '+' exists
768  }
769  else if (flags & FLAGS_SPACE) {
770  buf[len++] = ' ';
771  }
772  }

795 // Computes the base-10 logarithm of the input number - which must be an actual
796 // positive number (not infinity or NaN, nor a sub-normal)
797 static double log10_of_positive(double positive_number)
798 {
799  // The implementation follows David Gay (https://www.ampl.com/netlib/fp/dtoa.c).
800  //
801  // Since log_10 ( M * 2^x ) = log_10(M) + x , we can separate the components of
802  // our input number, and need only solve log_10(M) for M between 1 and 2 (as
803  // the base-2 mantissa is always 1-point-something). In that limited range, a
804  // Taylor series expansion of log10(x) should serve us well enough; and we'll
805  // take the mid-point, 1.5, as the point of expansion.
806 
807  double_with_bit_access dwba = get_bit_access(positive_number);
808  // based on the algorithm by David Gay (https://www.ampl.com/netlib/fp/dtoa.c)
809  int exp2 = get_exp2(dwba);
810  // drop the exponent, so dwba.F comes into the range [1,2)
811  dwba.U = (dwba.U & (((double_uint_t) (1) << DOUBLE_STORED_MANTISSA_BITS) - 1U)) |
812  ((double_uint_t) DOUBLE_BASE_EXPONENT << DOUBLE_STORED_MANTISSA_BITS);
813  double z = (dwba.F - 1.5);
814  return (
815  // Taylor expansion around 1.5:
816  0.1760912590556812420 // Expansion term 0: ln(1.5) / ln(10)
817  + z * 0.2895296546021678851 // Expansion term 1: (M - 1.5) * 2/3 / ln(10)
818 #if PRINTF_LOG10_TAYLOR_TERMS > 2
819  - z*z * 0.0965098848673892950 // Expansion term 2: (M - 1.5)^2 * 2/9 / ln(10)
820 #if PRINTF_LOG10_TAYLOR_TERMS > 3
821  + z*z*z * 0.0428932821632841311 // Expansion term 2: (M - 1.5)^3 * 8/81 / ln(10)
822 #endif
823 #endif
824  // exact log_2 of the exponent x, with logarithm base change
825  + exp2 * 0.30102999566398119521 // = exp2 * log_10(2) = exp2 * ln(2)/ln(10)
826  );
827 }

830 static double pow10_of_int(int floored_exp10)
831 {
832  // A crude hack for avoiding undesired behavior with barely-normal or slightly-subnormal values.
833  if (floored_exp10 == DOUBLE_MAX_SUBNORMAL_EXPONENT_OF_10) {
834  return DOUBLE_MAX_SUBNORMAL_POWER_OF_10;
835  }
836  // Compute 10^(floored_exp10) but (try to) make sure that doesn't overflow
837  double_with_bit_access dwba;
838  int exp2 = bastardized_floor(floored_exp10 * 3.321928094887362 + 0.5);
839  const double z = floored_exp10 * 2.302585092994046 - exp2 * 0.6931471805599453;
840  const double z2 = z * z;
841  dwba.U = ((double_uint_t)(exp2) + DOUBLE_BASE_EXPONENT) << DOUBLE_STORED_MANTISSA_BITS;
842  // compute exp(z) using continued fractions,
843  // see https://en.wikipedia.org/wiki/Exponential_function#Continued_fractions_for_ex
844  dwba.F *= 1 + 2 * z / (2 - z + (z2 / (6 + (z2 / (10 + z2 / 14)))));
845  return dwba.F;
846 }

848 static void print_exponential_number(output_gadget_t* output, double number, printf_size_t precision, printf_size_t width, printf_flags_t flags, char* buf, printf_size_t len)
849 {
850  const bool negative = get_sign_bit(number);
851  // This number will decrease gradually (by factors of 10) as we "extract" the exponent out of it
852  double abs_number = negative ? -number : number;
853 
854  int floored_exp10;
855  bool abs_exp10_covered_by_powers_table;
856  struct scaling_factor normalization;
857 
858 
859  // Determine the decimal exponent
860  if (abs_number == 0.0) {
861  // TODO: This is a special-case for 0.0 (and -0.0); but proper handling is required for denormals more generally.
862  floored_exp10 = 0; // ... and no need to set a normalization factor or check the powers table
863  }
864  else {
865  double exp10 = log10_of_positive(abs_number);
866  floored_exp10 = bastardized_floor(exp10);
867  double p10 = pow10_of_int(floored_exp10);
868  // correct for rounding errors
869  if (abs_number < p10) {
870  floored_exp10--;
871  p10 /= 10;
872  }
873  abs_exp10_covered_by_powers_table = PRINTF_ABS(floored_exp10) < PRINTF_MAX_PRECOMPUTED_POWER_OF_10;
874  normalization.raw_factor = abs_exp10_covered_by_powers_table ? powers_of_10[PRINTF_ABS(floored_exp10)] : p10;
875  }
876 
877  // We now begin accounting for the widths of the two parts of our printed field:
878  // the decimal part after decimal exponent extraction, and the base-10 exponent part.
879  // For both of these, the value of 0 has a special meaning, but not the same one:
880  // a 0 exponent-part width means "don't print the exponent"; a 0 decimal-part width
881  // means "use as many characters as necessary".
882 
883  bool fall_back_to_decimal_only_mode = false;
884  if (flags & FLAGS_ADAPT_EXP) {
885  int required_significant_digits = (precision == 0) ? 1 : (int) precision;
886  // Should we want to fall-back to "%f" mode, and only print the decimal part?
887  fall_back_to_decimal_only_mode = (floored_exp10 >= -4 && floored_exp10 < required_significant_digits);
888  // Now, let's adjust the precision
889  // This also decided how we adjust the precision value - as in "%g" mode,
890  // "precision" is the number of _significant digits_, and this is when we "translate"
891  // the precision value to an actual number of decimal digits.
892  int precision_ = fall_back_to_decimal_only_mode ?
893  (int) precision - 1 - floored_exp10 :
894  (int) precision - 1; // the presence of the exponent ensures only one significant digit comes before the decimal point
895  precision = (precision_ > 0 ? (unsigned) precision_ : 0U);
896  flags |= FLAGS_PRECISION; // make sure print_broken_up_decimal respects our choice above
897  }
898 
899  normalization.multiply = (floored_exp10 < 0 && abs_exp10_covered_by_powers_table);
900  bool should_skip_normalization = (fall_back_to_decimal_only_mode || floored_exp10 == 0);
901  struct double_components decimal_part_components =
902  should_skip_normalization ?
903  get_components(negative ? -abs_number : abs_number, precision) :
904  get_normalized_components(negative, precision, abs_number, normalization, floored_exp10);
905 
906  // Account for roll-over, e.g. rounding from 9.99 to 100.0 - which effects
907  // the exponent and may require additional tweaking of the parts
908  if (fall_back_to_decimal_only_mode) {
909  if ((flags & FLAGS_ADAPT_EXP) && floored_exp10 >= -1 && decimal_part_components.integral == powers_of_10[floored_exp10 + 1]) {
910  floored_exp10++; // Not strictly necessary, since floored_exp10 is no longer really used
911  precision--;
912  // ... and it should already be the case that decimal_part_components.fractional == 0
913  }
914  // TODO: What about rollover strictly within the fractional part?
915  }
916  else {
917  if (decimal_part_components.integral >= 10) {
918  floored_exp10++;
919  decimal_part_components.integral = 1;
920  decimal_part_components.fractional = 0;
921  }
922  }
923 
924  // the floored_exp10 format is "E%+03d" and largest possible floored_exp10 value for a 64-bit double
925  // is "307" (for 2^1023), so we set aside 4-5 characters overall
926  printf_size_t exp10_part_width = fall_back_to_decimal_only_mode ? 0U : (PRINTF_ABS(floored_exp10) < 100) ? 4U : 5U;
927 
928  printf_size_t decimal_part_width =
929  ((flags & FLAGS_LEFT) && exp10_part_width) ?
930  // We're padding on the right, so the width constraint is the exponent part's
931  // problem, not the decimal part's, so we'll use as many characters as we need:
932  0U :
933  // We're padding on the left; so the width constraint is the decimal part's
934  // problem. Well, can both the decimal part and the exponent part fit within our overall width?
935  ((width > exp10_part_width) ?
936  // Yes, so we limit our decimal part's width.
937  // (Note this is trivially valid even if we've fallen back to "%f" mode)
938  width - exp10_part_width :
939  // No; we just give up on any restriction on the decimal part and use as many
940  // characters as we need
941  0U);
942 
943  const printf_size_t printed_exponential_start_pos = output->pos;
944  print_broken_up_decimal(decimal_part_components, output, precision, decimal_part_width, flags, buf, len);
945 
946  if (! fall_back_to_decimal_only_mode) {
947  putchar_via_gadget(output, (flags & FLAGS_UPPERCASE) ? 'E' : 'e');
948  print_integer(output,
949  ABS_FOR_PRINTING(floored_exp10),
950  floored_exp10 < 0, 10, 0, exp10_part_width - 1,
951  FLAGS_ZEROPAD | FLAGS_PLUS);
952  if (flags & FLAGS_LEFT) {
953  // We need to right-pad with spaces to meet the width requirement
954  while (output->pos - printed_exponential_start_pos < width) {
955  putchar_via_gadget(output, ' ');
956  }
957  }
958  }
959 }
