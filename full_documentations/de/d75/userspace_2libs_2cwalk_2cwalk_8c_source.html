362 static bool cwk_path_segment_back_will_be_removed(struct cwk_segment_joined *sj)
363 {
364  enum cwk_segment_type type;
365  int counter;
366 
367  // We are handling back segments here. We must verify how many back segments
368  // and how many normal segments come before this one to decide whether we keep
369  // or remove it.
370 
371  // The counter determines how many normal segments are our current segment,
372  // which will popped off before us. If the counter goes above zero it means
373  // that our segment will be popped as well.
374  counter = 0;
375 
376  // We loop over all previous segments until we either reach the beginning,
377  // which means our segment will not be dropped or the counter goes above zero.
378  while (cwk_path_get_previous_segment_joined(sj)) {
379 
380  // Now grab the type. The type determines whether we will increase or
381  // decrease the counter. We don't handle a CWK_CURRENT frame here since it
382  // has no influence.
383  type = cwk_path_get_segment_type(&sj->segment);
384  if (type == CWK_NORMAL) {
385  // This is a normal segment. The normal segment will increase the counter
386  // since it neutralizes one back segment. If we go above zero we can
387  // return immediately.
388  ++counter;
389  if (counter > 0) {
390  return true;
391  }
392  } else if (type == CWK_BACK) {
393  // A CWK_BACK segment will reduce the counter by one. We can not remove a
394  // back segment as long we are not above zero since we don't have the
395  // opposite normal segment which we would remove.
396  --counter;
397  }
398  }
399 
400  // We never got a count larger than zero, so we will keep this segment alive.
401  return false;

476 static void cwk_path_get_root_windows(const char *path, size_t *length)
477 {
478  const char *c;
479  bool is_device_path;
480 
481  // We can not determine the root if this is an empty string. So we set the
482  // root to NULL and the length to zero and cancel the whole thing.
483  c = path;
484  *length = 0;
485  if (!*c) {
486  return;
487  }
488 
489  // Now we have to verify whether this is a windows network path (UNC), which
490  // we will consider our root.
491  if (cwk_path_is_separator(c)) {
492  ++c;
493 
494  // Check whether the path starts with a single backslash, which means this
495  // is not a network path - just a normal path starting with a backslash.
496  if (!cwk_path_is_separator(c)) {
497  // Okay, this is not a network path but we still use the backslash as a
498  // root.
499  ++(*length);
500  return;
501  }
502 
503  // A device path is a path which starts with "\\." or "\\?". A device path
504  // can be a UNC path as well, in which case it will take up one more
505  // segment. So, this is a network or device path. Skip the previous
506  // separator. Now we need to determine whether this is a device path. We
507  // might advance one character here if the server name starts with a '?' or
508  // a '.', but that's fine since we will search for a separator afterwards
509  // anyway.
510  ++c;
511  is_device_path = (*c == '?' || *c == '.') && cwk_path_is_separator(++c);
512  if (is_device_path) {
513  // That's a device path, and the root must be either "\\.\" or "\\?\"
514  // which is 4 characters long. (at least that's how Windows
515  // GetFullPathName behaves.)
516  *length = 4;
517  return;
518  }
519 
520  // We will grab anything up to the next stop. The next stop might be a '\0'
521  // or another separator. That will be the server name.
522  c = cwk_path_find_next_stop(c);
523 
524  // If this is a separator and not the end of a string we wil have to include
525  // it. However, if this is a '\0' we must not skip it.
526  while (cwk_path_is_separator(c)) {
527  ++c;
528  }
529 
530  // We are now skipping the shared folder name, which will end after the
531  // next stop.
532  c = cwk_path_find_next_stop(c);
533 
534  // Then there might be a separator at the end. We will include that as well,
535  // it will mark the path as absolute.
536  if (cwk_path_is_separator(c)) {
537  ++c;
538  }
539 
540  // Finally, calculate the size of the root.
541  *length = (size_t)(c - path);
542  return;
543  }
544 
545  // Move to the next and check whether this is a colon.
546  if (*++c == ':') {
547  *length = 2;
548 
549  // Now check whether this is a backslash (or slash). If it is not, we could
550  // assume that the next character is a '\0' if it is a valid path. However,
551  // we will not assume that - since ':' is not valid in a path it must be a
552  // mistake by the caller than. We will try to understand it anyway.
553  if (cwk_path_is_separator(++c)) {
554  *length = 3;
555  }
556  }

688  // correctly. However, he might not and in that case we will append a fake
689  // root at the beginning.
690  if (cwk_path_is_absolute(base)) {
691  i = 0;
692  } else if (path_style == CWK_STYLE_WINDOWS) {
693  paths[0] = "\\";
694  i = 1;
695  } else {
696  paths[0] = "/";
697  i = 1;
698  }
699

701  // If the submitted path is not relative the base path becomes irrelevant.
702  // We will only normalize the submitted path instead.
703  paths[i++] = path;

705  } else {

707  // The result will be a new absolute path.

714  return cwk_path_join_and_normalize_multiple(paths, buffer, buffer_size);

717 static void cwk_path_skip_segments_until_diverge(struct cwk_segment_joined *bsj,
718  struct cwk_segment_joined *osj, bool absolute, bool *base_available,
719  bool *other_available)
720 {
721  // Now looping over all segments until they start to diverge. A path may
722  // diverge if two segments are not equal or if one path reaches the end.
723  do {
724 
725  // Check whether there is anything available after we skip everything which
726  // is invisible. We do that for both paths, since we want to let the caller
727  // know which path has some trailing segments after they diverge.
728  *base_available = cwk_path_segment_joined_skip_invisible(bsj, absolute);
729  *other_available = cwk_path_segment_joined_skip_invisible(osj, absolute);
730 
731  // We are done if one or both of those paths reached the end. They either
732  // diverge or both reached the end - but in both cases we can not continue
733  // here.
734  if (!*base_available || !*other_available) {
735  break;
736  }
737 
738  // Compare the content of both segments. We are done if they are not equal,
739  // since they diverge.
740  if (!cwk_path_is_string_equal(bsj->segment.begin, osj->segment.begin,
741  bsj->segment.size, osj->segment.size)) {
742  break;
743  }
744 
745  // We keep going until one of those segments reached the end. The next
746  // segment might be invisible, but we will check for that in the beginning
747  // of the loop once again.
748  *base_available = cwk_path_get_next_segment_joined(bsj);
749  *other_available = cwk_path_get_next_segment_joined(osj);
750  } while (*base_available && *other_available);
751 }

753 size_t cwk_path_get_relative(const char *base_directory, const char *path,
754  char *buffer, size_t buffer_size)
755 {
756  size_t pos, base_root_length, path_root_length;
757  bool absolute, base_available, other_available, has_output;
758  const char *base_paths[2], *other_paths[2];
759  struct cwk_segment_joined bsj, osj;
760 
761  pos = 0;
762 
763  // First we compare the roots of those two paths. If the roots are not equal
764  // we can't continue, since there is no way to get a relative path from
765  // different roots.
766  cwk_path_get_root(base_directory, &base_root_length);
767  cwk_path_get_root(path, &path_root_length);
768  if (base_root_length != path_root_length ||
769  !cwk_path_is_string_equal(base_directory, path, base_root_length,
770  path_root_length)) {
771  cwk_path_terminate_output(buffer, buffer_size, pos);
772  return pos;
773  }
774 
775  // Verify whether this is an absolute path. We need to know that since we can
776  // remove all back-segments if it is.
777  absolute = cwk_path_is_root_absolute(base_directory, base_root_length);
778 
779  // Initialize our joined segments. This will allow us to use the internal
780  // functions to skip until diverge and invisible. We only have one path in
781  // them though.
782  base_paths[0] = base_directory;
783  base_paths[1] = NULL;
784  other_paths[0] = path;
785  other_paths[1] = NULL;
786  cwk_path_get_first_segment_joined(base_paths, &bsj);
787  cwk_path_get_first_segment_joined(other_paths, &osj);
788 
789  // Okay, now we skip until the segments diverge. We don't have anything to do
790  // with the segments which are equal.
791  cwk_path_skip_segments_until_diverge(&bsj, &osj, absolute, &base_available,
792  &other_available);
793 
794  // Assume there is no output until we have got some. We will need this
795  // information later on to remove trailing slashes or alternatively output a
796  // current-segment.
797  has_output = false;
798 
799  // So if we still have some segments left in the base path we will now output
800  // a back segment for all of them.
801  if (base_available) {
802  do {
803  // Skip any invisible segment. We don't care about those and we don't need
804  // to navigate back because of them.
805  if (!cwk_path_segment_joined_skip_invisible(&bsj, absolute)) {
806  break;
807  }
808 
809  // Toggle the flag if we have output. We need to remember that, since we
810  // want to remove the trailing slash.
811  has_output = true;
812 
813  // Output the back segment and a separator. No need to worry about the
814  // superfluous segment since it will be removed later on.
815  pos += cwk_path_output_back(buffer, buffer_size, pos);
816  pos += cwk_path_output_separator(buffer, buffer_size, pos);
817  } while (cwk_path_get_next_segment_joined(&bsj));
818  }
819 
820  // And if we have some segments available of the target path we will output
821  // all of those.
822  if (other_available) {
823  do {
824  // Again, skip any invisible segments since we don't need to navigate into
825  // them.
826  if (!cwk_path_segment_joined_skip_invisible(&osj, absolute)) {
827  break;
828  }
829 
830  // Toggle the flag if we have output. We need to remember that, since we
831  // want to remove the trailing slash.
832  has_output = true;
833 
834  // Output the current segment and a separator. No need to worry about the
835  // superfluous segment since it will be removed later on.
836  pos += cwk_path_output_sized(buffer, buffer_size, pos, osj.segment.begin,
837  osj.segment.size);
838  pos += cwk_path_output_separator(buffer, buffer_size, pos);
839  } while (cwk_path_get_next_segment_joined(&osj));
840  }
841 
842  // If we have some output by now we will have to remove the trailing slash. We
843  // simply do that by moving back one character. The terminate output function
844  // will then place the '\0' on this position. Otherwise, if there is no
845  // output, we will have to output a "current directory", since the target path
846  // points to the base path.
847  if (has_output) {
848  --pos;
849  } else {
850  pos += cwk_path_output_current(buffer, buffer_size, pos);
851  }
852 
853  // Finally, we can terminate the output - which means we place a '\0' at the
854  // current position or at the end of the buffer.
855  cwk_path_terminate_output(buffer, buffer_size, pos);
856 
857  return pos;
858 }

1076  char *buffer, size_t buffer_size)
1077 {
1078  struct cwk_segment segment;
1079  const char *c, *old_extension;
1080  size_t pos, root_size, trail_size, new_extension_size;
1081 
1082  // First we try to get the last segment. We may only have a root without any
1083  // segments, in which case we will create one.
1084  if (!cwk_path_get_last_segment(path, &segment)) {
1085 
1086  // So there is no segment in this path. First we grab the root and output
1087  // that. We are not going to modify the root in any way. If there is no
1088  // root, this will end up with a root size 0, and nothing will be written.
1089  cwk_path_get_root(path, &root_size);
1090  pos = cwk_path_output_sized(buffer, buffer_size, 0, path, root_size);
1091 
1092  // Add a dot if the submitted value doesn't have any.
1093  if (*new_extension != '.') {
1094  pos += cwk_path_output_dot(buffer, buffer_size, pos);
1095  }
1096 
1097  // And finally terminate the output and return the total size of the path.
1098  pos += cwk_path_output(buffer, buffer_size, pos, new_extension);
1099  cwk_path_terminate_output(buffer, buffer_size, pos);
1100  return pos;
1101  }
1102 
1103  // Now we seek the old extension in the last segment, which we will replace
1104  // with the new one. If there is no old extension, it will point to the end of
1105  // the segment.
1106  old_extension = segment.end;
1107  for (c = segment.begin; c < segment.end; ++c) {
1108  if (*c == '.') {
1109  old_extension = c;
1110  }
1111  }
1112 
1113  pos = cwk_path_output_sized(buffer, buffer_size, 0, segment.path,
1114  (size_t)(old_extension - segment.path));
1115 
1116  // If the new extension starts with a dot, we will skip that dot. We always
1117  // output exactly one dot before the extension. If the extension contains
1118  // multiple dots, we will output those as part of the extension.
1119  if (*new_extension == '.') {
1120  ++new_extension;
1121  }
1122 
1123  // We calculate the size of the new extension, including the dot, in order to
1124  // output the trail - which is any part of the path coming after the
1125  // extension. We must output this first, since the buffer may overlap with the
1126  // submitted path - and it would be overridden by longer extensions.
1127  new_extension_size = strlen(new_extension) + 1;
1128  trail_size = cwk_path_output(buffer, buffer_size, pos + new_extension_size,
1129  segment.end);
1130 
1131  // Finally we output the dot and the new extension. The new extension itself
1132  // doesn't contain the dot anymore, so we must output that first.
1133  pos += cwk_path_output_dot(buffer, buffer_size, pos);
1134  pos += cwk_path_output(buffer, buffer_size, pos, new_extension);
1135 
1136  // Now we terminate the output with a null-terminating character, but before
1137  // we do that we must add the size of the trail to the position which we
1138  // output before.
1139  pos += trail_size;
1140  cwk_path_terminate_output(buffer, buffer_size, pos);
1141 
1142  // And the position is our output size now.
1143  return pos;
1144 }

1158 size_t cwk_path_get_intersection(const char *path_base, const char *path_other)
1159 {

1161  size_t base_root_length, other_root_length;
1162  const char *end;
1163  const char *paths_base[2], *paths_other[2];
1164  struct cwk_segment_joined base, other;
1165 
1166  // We first compare the two roots. We just return zero if they are not equal.
1167  // This will also happen to return zero if the paths are mixed relative and
1168  // absolute.
1169  cwk_path_get_root(path_base, &base_root_length);
1170  cwk_path_get_root(path_other, &other_root_length);
1171  if (!cwk_path_is_string_equal(path_base, path_other, base_root_length,
1172  other_root_length)) {
1173  return 0;
1174  }
1175 
1176  // Configure our paths. We just have a single path in here for now.
1177  paths_base[0] = path_base;
1178  paths_base[1] = NULL;
1179  paths_other[0] = path_other;
1180  paths_other[1] = NULL;
1181 
1182  // So we get the first segment of both paths. If one of those paths don't have
1183  // any segment, we will return 0.
1184  if (!cwk_path_get_first_segment_joined(paths_base, &base) ||
1185  !cwk_path_get_first_segment_joined(paths_other, &other)) {
1186  return base_root_length;
1187  }
1188 
1189  // We now determine whether the path is absolute or not. This is required
1190  // because if will ignore removed segments, and this behaves differently if
1191  // the path is absolute. However, we only need to check the base path because
1192  // we are guaranteed that both paths are either relative or absolute.
1193  absolute = cwk_path_is_root_absolute(path_base, base_root_length);
1194 
1195  // We must keep track of the end of the previous segment. Initially, this is
1196  // set to the beginning of the path. This means that 0 is returned if the
1197  // first segment is not equal.
1198  end = path_base + base_root_length;
1199 
1200  // Now we loop over both segments until one of them reaches the end or their
1201  // contents are not equal.
1202  do {
1203  // We skip all segments which will be removed in each path, since we want to
1204  // know about the true path.
1205  if (!cwk_path_segment_joined_skip_invisible(&base, absolute) ||
1206  !cwk_path_segment_joined_skip_invisible(&other, absolute)) {
1207  break;
1208  }
1209 
1210  if (!cwk_path_is_string_equal(base.segment.begin, other.segment.begin,
1211  base.segment.size, other.segment.size)) {
1212  // So the content of those two segments are not equal. We will return the
1213  // size up to the beginning.
1214  return (size_t)(end - path_base);
1215  }
1216 
1217  // Remember the end of the previous segment before we go to the next one.
1218  end = base.segment.end;
1219  } while (cwk_path_get_next_segment_joined(&base) &&
1220  cwk_path_get_next_segment_joined(&other));
1221 
1222  // Now we calculate the length up to the last point where our paths pointed to
1223  // the same place.
1224  return (size_t)(end - path_base);
1225 }

1362  char *buffer, size_t buffer_size)
1363 {
1364  size_t pos, value_size, tail_size;
1365 
1366  // First we have to output the head, which is the whole string up to the
1367  // beginning of the segment. This part of the path will just stay the same.
1368  pos = cwk_path_output_sized(buffer, buffer_size, 0, segment->path,
1369  (size_t)(segment->begin - segment->path));
1370 
1371  // In order to trip the submitted value, we will skip any separator at the
1372  // beginning of it and behave as if it was never there.
1373  while (cwk_path_is_separator(value)) {
1374  ++value;
1375  }
1376 
1377  // Now we determine the length of the value. In order to do that we first
1378  // locate the '\0'.
1379  value_size = 0;
1380  while (value[value_size]) {
1381  ++value_size;
1382  }
1383 
1384  // Since we trim separators at the beginning and in the end of the value we
1385  // have to subtract from the size until there are either no more characters
1386  // left or the last character is no separator.
1387  while (value_size > 0 && cwk_path_is_separator(&value[value_size - 1])) {
1388  --value_size;
1389  }
1390 
1391  // We also have to determine the tail size, which is the part of the string
1392  // following the current segment. This part will not change.
1393  tail_size = strlen(segment->end);
1394 
1395  // Now we output the tail. We have to do that, because if the buffer and the
1396  // source are overlapping we would override the tail if the value is
1397  // increasing in length.
1398  cwk_path_output_sized(buffer, buffer_size, pos + value_size, segment->end,
1399  tail_size);
1400 
1401  // Finally we can output the value in the middle of the head and the tail,
1402  // where we have enough space to fit the whole trimmed value.
1403  pos += cwk_path_output_sized(buffer, buffer_size, pos, value, value_size);
1404 
1405  // Now we add the tail size to the current position and terminate the output -
1406  // basically, ensure that there is a '\0' at the end of the buffer.
1407  pos += tail_size;
1408  cwk_path_terminate_output(buffer, buffer_size, pos);
1409 
1410  // And now tell the caller how long the whole path would be.
1411  return pos;
1412 }

1415 {

1417  size_t root_length;
1418  struct cwk_segment segment;
1419

1421  // slash, so if we can determine that it starts with something like "C:", we
1422  // know that this is a windows path.
1423  cwk_path_get_root_windows(path, &root_length);
1424  if (root_length > 1) {
1425  return CWK_STYLE_WINDOWS;
1426  }
1427 
1428  // Next we check for slashes. Windows uses backslashes, while unix uses
1429  // forward slashes. Windows actually supports both, but our best guess is to
1430  // assume windows with backslashes and unix with forward slashes.
1431  for (c = path; *c; ++c) {
1432  if (*c == *separators[CWK_STYLE_UNIX]) {
1433  return CWK_STYLE_UNIX;
1434  } else if (*c == *separators[CWK_STYLE_WINDOWS]) {
1435  return CWK_STYLE_WINDOWS;
1436  }
1437  }
1438 
1439  // This path does not have any slashes. We grab the last segment (which
1440  // actually must be the first one), and determine whether the segment starts
1441  // with a dot. A dot is a hidden folder or file in the UNIX world, in that
1442  // case we assume the path to have UNIX style.
1443  if (!cwk_path_get_last_segment(path, &segment)) {
1444  // We couldn't find any segments, so we default to a UNIX path style since
1445  // there is no way to make any assumptions.
1446  return CWK_STYLE_UNIX;
1447  }
1448 
1449  if (*segment.begin == '.') {
1450  return CWK_STYLE_UNIX;
1451  }
1452 
1453  // And finally we check whether the last segment contains a dot. If it
1454  // contains a dot, that might be an extension. Windows is more likely to have
1455  // file names with extensions, so our guess would be windows.
1456  for (c = segment.begin; *c; ++c) {
1457  if (*c == '.') {
1458  return CWK_STYLE_WINDOWS;
1459  }
1460  }
1461 
1462  // All our checks failed, so we will return a default value which is currently
1463  // UNIX.
1464  return CWK_STYLE_UNIX;
